void push(int value) {
	node_s* node_ptr = new node_s;

	node_ptr->data = value; // записать данные

	node_ptr->previous_ptr = NULL; // предыдущего нет

	node_ptr->next_ptr = top_ptr; // следующий известен

	top_ptr = node_ptr; // теперь верхний ито этот

	if (counter == 0) { // то и нижний тоже
		buttom_ptr = node_ptr;
	}

	if (counter > 0) {
		node_ptr->next_ptr->previous_ptr = node_ptr; // предыдущий последующего ито этот
	}
	counter++;
}

int pop() {
	int rec = buttom_ptr->data; // результат ф-ции 
	node_s* tmp = buttom_ptr->previous_ptr;

	delete buttom_ptr;

	counter--;

	buttom_ptr = tmp;

	if (counter > 0) {
		buttom_ptr->next_ptr = NULL;
	}

	return rec;
}
